	sendSignal(type: Signal, connection?: Connection, data?: any): void {
		const signalOptions: SignalOptions = {

			to: connection ? [connection] : undefined

		if (type === Signal.NICKNAME_CHANGED && !!this.getScreenSession().connection) {
